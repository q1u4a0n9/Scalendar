<script>
        document.addEventListener('DOMContentLoaded', () => {
            const options = document.querySelectorAll('.option-card');
            const continueBtn = document.getElementById('continue-btn');
            let hasSelection = false;

            options.forEach(option => {
                option.addEventListener('click', () => {
                    // Deselect all
                    options.forEach(opt => opt.classList.remove('selected'));
                    
                    // Select clicked
                    option.classList.add('selected');
                    hasSelection = true;

                    // Update button state
                    if (hasSelection) {
                        continueBtn.classList.remove('bg-surface-variant', 'text-on-surface-variant', 'opacity-70', 'cursor-not-allowed');
                        continueBtn.classList.add('bg-primary', 'text-on-primary', 'shadow-md-soft', 'hover:bg-on-primary-fixed', 'opacity-100');
                    }
                });
            });
        });
